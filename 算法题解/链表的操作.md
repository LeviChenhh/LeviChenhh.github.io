# 链表的操作

## 19. 删除链表的倒数第 N 个结点（快慢指针）
给你一个链表，删除链表的倒数第 n 个结点，并且返回链表的头结点。
>https://leetcode-cn.com/problems/remove-nth-node-from-end-of-list/
借助一个头结点，否则处理长度为1的链表时会报错；快慢指针

```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        ListNode * hhead = new ListNode(0, head);
        ListNode *pFast = hhead, *pSlow = hhead;
        int cnt = 0;

        while(pFast != nullptr)
        {
            pFast = pFast->next;
            // 保证两者的距离等于n
            if(cnt < n + 2) cnt++;
            if(cnt == n + 2) pSlow = pSlow->next;
        }
        pSlow->next = pSlow->next->next;
        head = hhead->next;
        delete hhead;
        return head;
    }
};
```

## 21. 合并两个有序链表
将两个升序链表合并为一个新的 升序 链表并返回。新链表是通过拼接给定的两个链表的所有节点组成的。 
>https://leetcode-cn.com/problems/merge-two-sorted-lists/

```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        ListNode *preHead = new ListNode(0);
        ListNode *end = preHead;  // 需要动态维护
        while(list1 != nullptr && list2 != nullptr)
        {
            if(list1->val < list2->val)
            {
                end->next = list1;
                list1 = list1->next;
            }
            else
            {
                end->next = list2;
                list2 = list2->next;
            }
            end = end->next;
        }

        end->next = list1 != nullptr ? list1 : list2;
        ListNode *ans = preHead->next;
        delete preHead;
        return ans;
    }
};
```
## 206. 反转链表
> https://leetcode-cn.com/problems/reverse-linked-list/

思路：使用
```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        ListNode* prev = nullptr;
        ListNode* curr = head;
        while (curr) {
            ListNode* next = curr->next;
            curr->next = prev;
            prev = curr;
            curr = next;
        }
        return prev;
    }
};
```
